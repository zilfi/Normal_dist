randomdeviates1 <- rnorm(100, mean=100, sd=15)  
randomdeviates2 <- rnorm(100, mean=100, sd=15)  

x1 <- seq(min(randomdeviates1), max(randomdeviates1), length=100)#, and graphs the p. d. f. of the same normal distribution (third and forth lines).
x2 <- seq(min(randomdeviates2), max(randomdeviates2), length=100)#, and graphs the p. d. f. of the same normal distribution (third and forth lines).


mu <- c(100,100)
sigma <- matrix(c(15,0,0,15),2,2)                                

f<-function(x1,x2){dmnorm(cbind(x1,x2), mu, sigma)}
f1<-function(x1,x2){pmnorm(cbind(x1,x2), mu, sigma)}

z<-outer(x1,x2,f)
z1<-outer(x1,x2,f1)

persp(x1,x2,z, ticktype="detailed", theta=40,phi=0)

persp(x1,x2,z, theta=100,phi=40,col="blue")

persp(x1,x2,z1, theta=100,phi=40,col="blue")

wireframe(z, colorkey = TRUE,  drape = TRUE)

wireframe(z1, colorkey = TRUE,  drape = TRUE)
